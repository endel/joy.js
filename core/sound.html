Source code

var engine = new Joy.Engine(document.getElementById('example'));
engine.createScene(function(scene) {
  var spock = new Joy.Sound({
    urls: [
      '/assets/examples/sound/Spock_Livelong.mp3',
      '/assets/examples/sound/Spock_Livelong.ogg',
      '/assets/examples/sound/Spock_Livelong.wav'
    ]
  });

  // Setup button
  var button = new Joy.Circle({
    x: engine.width / 2 - 100,
    y: engine.height / 2 - 100,
    radius: 100,
    color: "#000"
  }).bind(Joy.Events.CLICK, function () {
    spock.play();
  });

  scene.addChild(button);

  // Add label text
  scene.addChild(new Joy.Text({
    x: button.position.x + button.width / 2,
    y: button.position.y + button.height / 2,
    align: "center",
    baseline: Joy.Text.BASELINE.MIDDLE,
    text: "Live long and prosper.",
    color: "#fff"
  }));
});
